<script>
const streams = {
  willow: "https://crichd.workspace-sultanarabi161.workers.dev/live/willowusa.m3u8",
  asport: "https://crichd.workspace-sultanarabi161.workers.dev/live/asportshd.m3u8",
  ss: "https://playztv-apps.pages.dev/star-sports/index.m3u8",
  nep: "https://embedsports.me/t20/nepal-vs-united-arab-emirates-stream-1"
};

const params = new URLSearchParams(window.location.search);
const src = params.get("src");
const id = params.get("id");
const url = src || (id ? streams[id] : null);

const video = document.getElementById("videoPlayer");
const iframe = document.getElementById("iframeViewer");
const loader = document.getElementById("loader");

const hideLoader = () => loader.style.display = "none";

if (!url) {
  document.body.innerHTML = "<h2 style='color:white;text-align:center;margin-top:40vh'>‚ùå Stream not available</h2>";
} else {
  const isHLS = url.toLowerCase().includes(".m3u8");
  const videoExt = ["mp4","webm","ogg","mov","mkv"];
  const isVideo = videoExt.some(ext => url.toLowerCase().includes(ext));

  if (isHLS || isVideo) {
    video.style.display = "block";
    if (isHLS && Hls.isSupported()) {
      const hls = new Hls();
      hls.loadSource(url);
      hls.attachMedia(video);
      hls.on(Hls.Events.MANIFEST_PARSED, () => { video.play(); hideLoader(); });
    } else {
      video.src = url;
      video.addEventListener("loadeddata", hideLoader);
    }
  } else {
    iframe.style.display = "block";
    iframe.src = url;
    iframe.onload = hideLoader;
    setTimeout(hideLoader, 6000);
  }
}
</script>
